24AuthenticationResult UserRepository::authenticate(const QString& username, const QString& password) const {
25 QSqlQuery query(Database::getInstance().getConnection());
26 query.prepare(R"(
27 SELECT id, nome, username, senha_hash, perfil, ativo
28 FROM usuarios
29 WHERE username = :username COLLATE NOCASE
30 )");
31 query.bindValue(":username", username.trimmed());
32
33 if (!query.exec() || !query.next()) {
34 return {AuthenticationStatus::InvalidCredentials, {}, "Usuário ou senha inválidos."};
35 }
36
37 const UserData user = userFromQuery(query);
38 if (!PasswordHasher::verify(password, query.value("senha_hash").toString())) {
39 return {AuthenticationStatus::InvalidCredentials, {}, "Usuário ou senha inválidos."};
40 }
41 if (!user.ativo) {
42 return {AuthenticationStatus::Inactive, {}, "Este usuário está desativado. Procure o usuário master."};
43 }
44
45 return {AuthenticationStatus::Success, user, {}};
46}

53 )");
54 query.bindValue(":nome", nome.trimmed());
55 query.bindValue(":username", username.trimmed().toLower());
56 query.bindValue(":senha_hash", PasswordHasher::hash(password));
57 query.bindValue(":criado_em", QDateTime::currentDateTime().toString(Qt::ISODate));
58
59 if (!query.exec()) {

61 }
