157void MarmoManagerApp::loadOSData() {
158 QString buscaCliente = m_searchInput->text().trimmed();
159 QString buscaStatus = m_searchStatus->currentText();
160 QString dataIni = "";
161 QString dataFim = "";
162
163 if (m_searchUseDate->isChecked()) {
164 dataIni = m_searchDataInicio->date().toString(Qt::ISODate);
165 dataFim = m_searchDataFim->date().toString(Qt::ISODate);
166 }
167
168 QVector<OSData> rows = m_controller.listarOS(buscaCliente, buscaStatus, dataIni, dataFim);
169
170 m_osTable->setRowCount(rows.size());
171 for (int i = 0; i < rows.size(); ++i) {
172 const auto& os = rows[i];
173 m_osTable->setItem(i, 0, new QTableWidgetItem(QString::number(os.id)));
174 m_osTable->setItem(i, 1, new QTableWidgetItem(os.cliente));
175 m_osTable->setItem(i, 2, new QTableWidgetItem(os.contato));
176 m_osTable->setItem(i, 3, new QTableWidgetItem(os.medidas));
177 m_osTable->setItem(i, 4, new QTableWidgetItem(os.material));
178 m_osTable->setItem(i, 5, new QTableWidgetItem(os.dataAbertura));
179 m_osTable->setItem(i, 6, new QTableWidgetItem(os.prazo));
180 m_osTable->setItem(i, 7, new QTableWidgetItem(os.status));
181 }
