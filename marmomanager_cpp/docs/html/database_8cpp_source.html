31QSqlDatabase Database::getConnection() const {
32 // `database()` abre a conexão automaticamente por padrão. Como o nome do
33 // arquivo precisa ser definido antes do open(), solicitamos explicitamente
34 // a conexão fechada para evitar que o SQLite use um banco temporário em memória.
35 QSqlDatabase db = QSqlDatabase::database(QSqlDatabase::defaultConnection, false);
36 if (db.databaseName() != m_dbName) {
37 if (db.isOpen()) {
38 db.close();
39 }
40 db.setDatabaseName(m_dbName);
41 }
42 if (!db.isOpen()) {
43 if (!db.open()) {
44 qWarning() << "Failed to open database:" << db.lastError().text();
45 }
46 }
47 if (db.isOpen()) {
48 QSqlQuery pragma(db);
49 if (!pragma.exec("PRAGMA foreign_keys = ON")) {
50 qWarning() << "Failed to enable SQLite foreign keys:" << pragma.lastError().text();
51 }
52 }
53 return db;
54}
